CREATE TABLE PRODUCT
(PRODID INTEGER NOT NULL IDENTITY,
NAME VARCHAR(32) NOT NULL,
DESCRIPTION VARCHAR(32))

CREATE TABLE CATEGORY
(CATID INTEGER NOT NULL IDENTITY,
NAME VARCHAR(32) NOT NULL,
DESCRIPTION VARCHAR(32))

CREATE TABLE PUBLIC.PRODCAT
(ID INTEGER NOT NULL IDENTITY,
CATID INTEGER NOT NULL,
PRODID INTEGER NOT NULL,
FOREIGN KEY (CATID) REFERENCES CATEGORY(CATID) ON UPDATE CASCADE ON DELETE NO ACTION,
FOREIGN KEY (PRODID) REFERENCES PRODUCT(PRODID) ON UPDATE CASCADE ON DELETE NO ACTION)
